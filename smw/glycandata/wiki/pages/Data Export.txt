
== Permethylated Molecular Weight ==

<pre>
PREFIX glycandata: <http://glycandata.glygen.org/glycandata#>

SELECT ?accession ?type ?property ?value ?source 
WHERE {
  ?sub a glycandata:Annotation .
  ?sub glycandata:type ?type .
  ?sub glycandata:property ?property .
  ?sub glycandata:value ?value .
  ?sub glycandata:source ?source .
  ?sub glycandata:hasglycan ?glycan .
  ?glycan a glycandata:Glycan . 
  ?glycan glycandata:accession ?accession .
  ?sub glycandata:property "PermethylatedMW" .
  ?sub glycandata:type "MolWt" . 
  ?sub glycandata:source "EdwardsLab" 
}
